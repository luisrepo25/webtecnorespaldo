--

CREATE FUNCTION public.cerrar_calificacion_final() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
DECLARE
    tiene_p1    INT;
    tiene_p2    INT;
    tiene_final INT;
    tiene_otros INT;
    suma_ponderada DECIMAL(5,2);
BEGIN
    SELECT
        COUNT(*) FILTER (WHERE tipo = 'parcial1'),
        COUNT(*) FILTER (WHERE tipo = 'parcial2'),
        COUNT(*) FILTER (WHERE tipo = 'final'),
        COUNT(*) FILTER (WHERE tipo = 'otros')
    INTO tiene_p1, tiene_p2, tiene_final, tiene_otros
    FROM evaluaciones
    WHERE id_inscripcion = NEW.id_inscripcion;

    IF tiene_p1 = 1 AND tiene_p2 = 1 AND tiene_final = 1 AND tiene_otros = 1 THEN
        SELECT ROUND(
            SUM(CASE WHEN tipo = 'parcial1' THEN calificacion * 0.25 ELSE 0 END) +
            SUM(CASE WHEN tipo = 'parcial2' THEN calificacion * 0.25 ELSE 0 END) +
            SUM(CASE WHEN tipo = 'final'    THEN calificacion * 0.40 ELSE 0 END) +
            SUM(CASE WHEN tipo = 'otros'    THEN calificacion * 0.10 ELSE 0 END),
        2)
        INTO suma_ponderada
        FROM evaluaciones
        WHERE id_inscripcion = NEW.id_inscripcion;

        UPDATE inscripciones
        SET calificacion_final = suma_ponderada,
            aprobado           = (suma_ponderada >= 51.00)
        WHERE id_inscripcion = NEW.id_inscripcion;
    END IF;

    RETURN NEW;
END;
$$;

CREATE FUNCTION public.crear_cuotas_credito() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
DECLARE
    total_materias    INT;
    costo_por_materia DECIMAL(10,2);
    mat_cubiertas     INT;
    mat_con_cuota     INT;
    restante          DECIMAL(10,2);
    monto_cuota       DECIMAL(10,2);
    i                 INT;
BEGIN
    IF NEW.forma_pago = 'credito' THEN
        SELECT COUNT(mc.id_malla) INTO total_materias
        FROM malla_curricular mc WHERE mc.id_carrera = NEW.id_carrera;

        IF total_materias IS NULL OR total_materias = 0 THEN
            total_materias := 1;
        END IF;

        costo_por_materia := ROUND(NEW.monto_total / total_materias, 2);
        mat_cubiertas := FLOOR(COALESCE(NEW.monto_pagado, 0) / costo_por_materia);
        IF mat_cubiertas > total_materias THEN mat_cubiertas := total_materias; END IF;

        mat_con_cuota := total_materias - mat_cubiertas;
        restante      := NEW.monto_total - COALESCE(NEW.monto_pagado, 0);

        IF mat_con_cuota > 0 AND restante > 0 THEN
            monto_cuota := ROUND(restante / mat_con_cuota, 2);
            FOR i IN 1..mat_con_cuota LOOP
                INSERT INTO cuotas_carrera
                    (id_pago_carrera, numero_cuota, monto, fecha_vencimiento, estado)
                VALUES (NEW.id_pago_carrera, i, monto_cuota,
                        CURRENT_DATE + (INTERVAL '1 month' * (mat_cubiertas + i)), 'pendiente');
            END LOOP;
        END IF;

        UPDATE pago_carrera_completa
        SET cuotas_total = mat_con_cuota, materias_cubiertas = mat_cubiertas
        WHERE id_pago_carrera = NEW.id_pago_carrera;
    END IF;
    RETURN NEW;
END;
$$;

--

CREATE FUNCTION public.crear_pago_inscripcion() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
DECLARE
    carrera_pagada  BOOLEAN;
    materia_actual  INT;
    carrera_est     INT;
    intentos_prev   INT;
    pago_carrera_id INT;
    costo_materia   DECIMAL(10,2);
BEGIN
    SELECT oa.id_materia INTO materia_actual
    FROM oferta_academica oa WHERE oa.id_oferta = NEW.id_oferta;

    SELECT costo_mensual INTO costo_materia
    FROM materias WHERE id_materia = materia_actual;

    SELECT e.id_carrera_actual INTO carrera_est
    FROM estudiantes e WHERE e.id_estudiante = NEW.id_estudiante;

    IF carrera_est IS NOT NULL THEN
        SELECT EXISTS(
            SELECT 1 FROM pago_carrera_completa pcc
            WHERE pcc.id_estudiante = NEW.id_estudiante
              AND pcc.id_carrera = carrera_est
              AND pcc.estado = 'pagado'
        ) INTO carrera_pagada;

        IF carrera_pagada = TRUE THEN
            SELECT pcc.id_pago_carrera INTO pago_carrera_id
            FROM pago_carrera_completa pcc
            WHERE pcc.id_estudiante = NEW.id_estudiante
              AND pcc.id_carrera = carrera_est AND pcc.estado = 'pagado'
            LIMIT 1;

            SELECT COUNT(*) INTO intentos_prev
            FROM consumo_materias_carrera
            WHERE id_pago_carrera = pago_carrera_id AND id_materia = materia_actual;

            IF intentos_prev >= 1 THEN
                INSERT INTO pago_materia_suelta (id_inscripcion, monto_pagado, fecha_pago, estado)
                VALUES (NEW.id_inscripcion, costo_materia, CURRENT_DATE, 'pendiente');
                INSERT INTO consumo_materias_carrera
                    (id_pago_carrera, id_materia, id_inscripcion, intento_numero, pagado_adicional)
                VALUES (pago_carrera_id, materia_actual, NEW.id_inscripcion, intentos_prev + 1, TRUE);
            ELSE
                INSERT INTO consumo_materias_carrera
                    (id_pago_carrera, id_materia, id_inscripcion, intento_numero)
                VALUES (pago_carrera_id, materia_actual, NEW.id_inscripcion, 1);
            END IF;
            RETURN NEW;
        END IF;
    END IF;

    INSERT INTO pago_materia_suelta (id_inscripcion, monto_pagado, fecha_pago, estado)
    VALUES (NEW.id_inscripcion, costo_materia, CURRENT_DATE, 'pendiente');
    RETURN NEW;
END;
$$;

-- Name: fn_confirmar_pago_qr(); Type: FUNCTION; Schema: public; Owner: postgres
--

CREATE FUNCTION public.fn_confirmar_pago_qr() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
DECLARE
    v_id_inscripcion INT;
    v_id_carrera     INT;
    v_costo_total    DECIMAL(10,2);
    v_costo_contado  DECIMAL(10,2);
    v_monto_total_registrar DECIMAL(10,2);
    v_forma_pago     VARCHAR(10);
    v_estado_carrera VARCHAR(15);
BEGIN
    IF NEW.estado = 'pagado' AND OLD.estado = 'pendiente' THEN
        NEW.fecha_pago := CURRENT_TIMESTAMP;

        -- CASO 1: Cobro de Matrícula Única
        IF NEW.concepto = 'matricula' THEN
            INSERT INTO matricula_unica (id_estudiante, monto_pagado, comprobante, estado)
            VALUES (NEW.id_estudiante, NEW.monto, NEW.payment_number, 'pagado')
            ON CONFLICT (id_estudiante) DO NOTHING;

        -- CASO 2: Cobro de Inscripción por Materia Suelta
        ELSIF NEW.concepto = 'materia' THEN
            IF NEW.id_inscripcion IS NOT NULL THEN
                v_id_inscripcion := NEW.id_inscripcion;
            ELSE
                SELECT i.id_inscripcion INTO v_id_inscripcion
                FROM inscripciones i
                JOIN grupos g ON i.id_oferta = g.id_oferta
                WHERE i.id_estudiante = NEW.id_estudiante
                  AND g.codigo_grupo = NEW.codigo_grupo
                  AND i.estado = 'pendiente_matricula'
                LIMIT 1;
            END IF;

            IF v_id_inscripcion IS NOT NULL THEN
                INSERT INTO pago_materia_suelta (id_inscripcion, monto_pagado, estado)
                VALUES (v_id_inscripcion, NEW.monto, 'pagado')
                ON CONFLICT (id_inscripcion) DO UPDATE
                SET estado = 'pagado', fecha_pago = CURRENT_DATE;

                UPDATE inscripciones SET estado = 'activo' WHERE id_inscripcion = v_id_inscripcion;
            ELSE
                RAISE NOTICE 'Advertencia: No se encontró inscripción pendiente para Estudiante % en Grupo %', NEW.id_estudiante, NEW.codigo_grupo;
            END IF;

        -- CASO 3: Pago de Carrera Completa (contado o parcial)
        -- codigo_grupo contiene el código de la carrera (ej: 'TEC-PROG')
        ELSIF NEW.concepto = 'carrera' THEN
            SELECT id_carrera, costo_carrera_completa
            INTO v_id_carrera, v_costo_total
            FROM carreras
            WHERE codigo = NEW.codigo_grupo;

            IF v_id_carrera IS NOT NULL THEN
                -- Descuento del 20% por pagar la carrera completa al contado (de una sola vez).
                -- El minimo de 30% para credito (validado en Java antes de generar el QR) sigue
                -- calculandose sobre el costo SIN descuento.
                v_costo_contado := ROUND(v_costo_total * 0.80, 2);

                IF NEW.monto >= v_costo_contado THEN
                    v_forma_pago         := 'contado';
                    v_estado_carrera     := 'pagado';
                    -- Contado con descuento: el total contractual queda en lo realmente pagado (sin deuda).
                    v_monto_total_registrar := NEW.monto;
                ELSE
                    v_forma_pago         := 'credito';
                    v_estado_carrera     := 'parcial';
                    -- Credito: el total a financiar sigue siendo el costo SIN descuento.
                    v_monto_total_registrar := v_costo_total;
                END IF;

                INSERT INTO pago_carrera_completa
                    (id_estudiante, id_carrera, monto_total, monto_pagado, forma_pago, estado, fecha_contrato)
                VALUES
                    (NEW.id_estudiante, v_id_carrera, v_monto_total_registrar, NEW.monto, v_forma_pago, v_estado_carrera, CURRENT_DATE)
                ON CONFLICT (id_estudiante, id_carrera) DO NOTHING;
            ELSE
                RAISE NOTICE 'Advertencia: No se encontró carrera con código % para pago QR', NEW.codigo_grupo;
            END IF;

        -- CASO 4: Pago de Cuota de Carrera
        -- id_inscripcion reutilizado como id_pago_carrera
        -- codigo_grupo   reutilizado como numero_cuota (string numérico)
        ELSIF NEW.concepto = 'cuota' THEN
            UPDATE cuotas_carrera
            SET    estado     = 'pagado',
                   fecha_pago = CURRENT_DATE
            WHERE  id_pago_carrera = NEW.id_inscripcion
              AND  numero_cuota    = NEW.codigo_grupo::INT
              AND  estado          = 'pendiente';

            UPDATE pago_carrera_completa
            SET estado = CASE
                WHEN (SELECT COUNT(*) FROM cuotas_carrera
                      WHERE id_pago_carrera = NEW.id_inscripcion
                        AND estado = 'pendiente') = 0
                     THEN 'pagado'
                ELSE 'parcial'
            END
            WHERE id_pago_carrera = NEW.id_inscripcion;
        END IF;
    END IF;
    RETURN NEW;
END;
$$;

CREATE FUNCTION public.fn_sync_seguimiento() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
BEGIN
    IF NEW.estado IN ('aprobado', 'reprobado') AND OLD.estado NOT IN ('aprobado', 'reprobado') THEN
        INSERT INTO seguimiento_estudiante
            (id_inscripcion, calificacion_final, estado_actual, fecha_aprobacion, created_at, updated_at)
        VALUES (
            NEW.id_inscripcion,
            NEW.calificacion_final,
            NEW.estado,
            COALESCE(NEW.fecha_aprobacion, CURRENT_DATE),
            NOW(), NOW()
        )
        ON CONFLICT (id_inscripcion) DO UPDATE
        SET calificacion_final = EXCLUDED.calificacion_final,
            estado_actual      = EXCLUDED.estado_actual,
            fecha_aprobacion   = EXCLUDED.fecha_aprobacion,
            updated_at         = NOW();
    END IF;
    RETURN NEW;
END;
$$;

CREATE FUNCTION public.marcar_cuota_al_pagar_materia() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
DECLARE
    pago_carrera_id INT;
BEGIN
    IF NEW.estado = 'pagado' AND OLD.estado = 'pendiente' THEN
        SELECT cmc.id_pago_carrera INTO pago_carrera_id
        FROM consumo_materias_carrera cmc
        WHERE cmc.id_inscripcion = NEW.id_inscripcion AND cmc.pagado_adicional = TRUE
        LIMIT 1;

        IF pago_carrera_id IS NOT NULL THEN
            UPDATE cuotas_carrera
            SET estado = 'pagado', fecha_pago = CURRENT_DATE
            WHERE id_cuota = (
                SELECT id_cuota FROM cuotas_carrera
                WHERE id_pago_carrera = pago_carrera_id AND estado = 'pendiente'
                ORDER BY numero_cuota ASC LIMIT 1
            );

            UPDATE pago_carrera_completa
            SET estado = CASE
                WHEN (SELECT COUNT(*) FROM cuotas_carrera
                      WHERE id_pago_carrera = pago_carrera_id AND estado = 'pendiente') = 0
                THEN 'pagado' ELSE 'parcial'
            END
            WHERE id_pago_carrera = pago_carrera_id;
        END IF;
    END IF;
    RETURN NEW;
END;
$$;

CREATE FUNCTION public.sincronizar_carrera_actual() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
BEGIN
    IF NEW.tipo_programa = 'carrera' AND NEW.estado = 'activo' THEN
        UPDATE estudiantes SET id_carrera_actual = NEW.id_carrera
        WHERE id_estudiante = NEW.id_estudiante;
    ELSIF OLD.tipo_programa = 'carrera' AND OLD.estado = 'activo' AND NEW.estado != 'activo' THEN
        UPDATE estudiantes SET id_carrera_actual = (
            SELECT id_carrera FROM afiliaciones_estudiante
            WHERE id_estudiante = NEW.id_estudiante
            AND tipo_programa = 'carrera' AND estado = 'activo'
            LIMIT 1
        ) WHERE id_estudiante = NEW.id_estudiante;
    END IF;
    RETURN NEW;
END;
$$;

CREATE FUNCTION public.validar_afiliacion_activa() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
DECLARE
    afiliacion_activa RECORD;
    materia_ofertada  INT;
    materia_en_malla  BOOLEAN;
BEGIN
    SELECT oa.id_materia INTO materia_ofertada
    FROM oferta_academica oa WHERE oa.id_oferta = NEW.id_oferta;

    SELECT * INTO afiliacion_activa
    FROM afiliaciones_estudiante
    WHERE id_estudiante = NEW.id_estudiante AND estado = 'activo'
    ORDER BY fecha_inicio DESC LIMIT 1;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'El estudiante no tiene una afiliación activa. Use "usuario registrar_estudiante" primero.';
    END IF;

    IF afiliacion_activa.tipo_programa = 'carrera' THEN
        SELECT EXISTS(
            SELECT 1 FROM malla_curricular mc
            JOIN ciclos c ON c.id_ciclos = mc.id_ciclo
            WHERE c.id_carrera = afiliacion_activa.id_carrera
              AND mc.id_materia = materia_ofertada
        ) INTO materia_en_malla;

        IF materia_en_malla = FALSE THEN
            RAISE EXCEPTION 'Esta materia no pertenece a la malla de la carrera del estudiante.';
        END IF;
    END IF;

    RETURN NEW;
END;
$$;

        FROM pago_carrera_completa pcc
        WHERE pcc.id_estudiante = NEW.id_estudiante
          AND pcc.id_carrera = carrera_estudiante
        LIMIT 1;

        IF pago_carrera_id IS NOT NULL THEN

            IF forma_pago_actual = 'credito' THEN
                SELECT COUNT(*) INTO consumos_actuales
                FROM consumo_materias_carrera WHERE id_pago_carrera = pago_carrera_id;

                SELECT materias_cubiertas INTO mat_cubiertas
                FROM pago_carrera_completa WHERE id_pago_carrera = pago_carrera_id;

                IF consumos_actuales < mat_cubiertas THEN
                    -- GRATIS
                    INSERT INTO consumo_materias_carrera
                        (id_pago_carrera, id_materia, id_inscripcion, intento_numero, pagado_adicional)
                    VALUES (pago_carrera_id, materia_actual, NEW.id_inscripcion, 1, FALSE);
                ELSE
                    SELECT monto INTO costo_materia
                    FROM cuotas_carrera
                    WHERE id_pago_carrera = pago_carrera_id AND estado = 'pendiente'
                    ORDER BY numero_cuota ASC LIMIT 1;

                    IF costo_materia IS NOT NULL THEN
                        INSERT INTO pago_materia_suelta
                            (id_inscripcion, monto_pagado, fecha_pago, estado)
                        VALUES (NEW.id_inscripcion, costo_materia, CURRENT_DATE, 'pendiente');
                        INSERT INTO consumo_materias_carrera
                            (id_pago_carrera, id_materia, id_inscripcion, intento_numero, pagado_adicional)
                        VALUES (pago_carrera_id, materia_actual, NEW.id_inscripcion, 1, TRUE);
                    ELSE
                        -- Sin cuotas pendientes: ya saldado, gratis
                        INSERT INTO consumo_materias_carrera
                            (id_pago_carrera, id_materia, id_inscripcion, intento_numero, pagado_adicional)
                        VALUES (pago_carrera_id, materia_actual, NEW.id_inscripcion, 1, FALSE);
                    END IF;
                END IF;

            ELSE -- CONTADO
                SELECT COUNT(*) INTO intentos_previos
                FROM consumo_materias_carrera
                WHERE id_pago_carrera = pago_carrera_id AND id_materia = materia_actual;

                IF intentos_previos = 0 THEN
                    INSERT INTO consumo_materias_carrera
                        (id_pago_carrera, id_materia, id_inscripcion, intento_numero, pagado_adicional)
                    VALUES (pago_carrera_id, materia_actual, NEW.id_inscripcion, 1, FALSE);
                ELSE
                    SELECT c.costo_carrera_completa, COUNT(mc.id_malla)
                    INTO costo_total_carrera, total_materias
                    FROM carreras c LEFT JOIN malla_curricular mc ON mc.id_carrera = c.id_carrera
                    WHERE c.id_carrera = carrera_estudiante GROUP BY c.costo_carrera_completa;

                    IF costo_total_carrera IS NOT NULL AND total_materias > 0 THEN
                        costo_materia := ROUND(costo_total_carrera / total_materias, 2);
                    ELSE
                        SELECT costo_mensual INTO costo_materia FROM materias WHERE id_materia = materia_actual;
                    END IF;

                    INSERT INTO pago_materia_suelta
                        (id_inscripcion, monto_pagado, fecha_pago, estado)
                    VALUES (NEW.id_inscripcion, costo_materia, CURRENT_DATE, 'pendiente');
                    INSERT INTO consumo_materias_carrera
                        (id_pago_carrera, id_materia, id_inscripcion, intento_numero, pagado_adicional)
                    VALUES (pago_carrera_id, materia_actual, NEW.id_inscripcion, intentos_previos + 1, TRUE);
                END IF;
            END IF;

        ELSE -- Sin pago de carrera
            SELECT c.costo_carrera_completa, COUNT(mc.id_malla)
            INTO costo_total_carrera, total_materias
            FROM carreras c LEFT JOIN malla_curricular mc ON mc.id_carrera = c.id_carrera
            WHERE c.id_carrera = carrera_estudiante GROUP BY c.costo_carrera_completa;

            IF costo_total_carrera IS NOT NULL AND total_materias > 0 THEN
                costo_materia := ROUND(costo_total_carrera / total_materias, 2);
            ELSE
                SELECT costo_mensual INTO costo_materia FROM materias WHERE id_materia = materia_actual;
            END IF;

            INSERT INTO pago_materia_suelta
                (id_inscripcion, monto_pagado, fecha_pago, estado)
            VALUES (NEW.id_inscripcion, costo_materia, CURRENT_DATE, 'pendiente');
        END IF;

    ELSE
        SELECT costo_mensual INTO costo_materia FROM materias WHERE id_materia = materia_actual;
        INSERT INTO pago_materia_suelta
            (id_inscripcion, monto_pagado, fecha_pago, estado)
        VALUES (NEW.id_inscripcion, costo_materia, CURRENT_DATE, 'pendiente');
    END IF;

    RETURN NEW;
END;
$$;


ALTER FUNCTION public.validar_pago_inscripcion() OWNER TO postgres;

--
-- Name: validar_requisito_aprobado(); Type: FUNCTION; Schema: public; Owner: postgres
--

CREATE FUNCTION public.validar_requisito_aprobado() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
DECLARE
    materia_requisito INT;
    requisito_aprobado BOOLEAN;
    nombre_requisito VARCHAR(150);
    nombre_materia VARCHAR(150);
BEGIN
    SELECT m.id_materia_requisito, m2.nombre, m.nombre
    INTO materia_requisito, nombre_requisito, nombre_materia
    FROM grupos oa
    JOIN materias m ON m.id_materia = oa.id_materia
    LEFT JOIN materias m2 ON m2.id_materia = m.id_materia_requisito
    WHERE oa.id_oferta = NEW.id_oferta;

    IF materia_requisito IS NULL THEN
        RETURN NEW;
    END IF;

    SELECT aprobado INTO requisito_aprobado
    FROM inscripciones i
    JOIN grupos oa ON oa.id_oferta = i.id_oferta
    WHERE i.id_estudiante = NEW.id_estudiante
    AND oa.id_materia = materia_requisito
    AND i.aprobado = TRUE
    ORDER BY i.id_inscripcion DESC
    LIMIT 1;
    
    IF requisito_aprobado IS DISTINCT FROM TRUE THEN
        RAISE EXCEPTION 'No puede inscribirse a "%". Debe aprobar primero "%"',
                        nombre_materia, COALESCE(nombre_requisito, 'materia requisito');
    END IF;
    
    RETURN NEW;
END;
$$;


ALTER FUNCTION public.validar_requisito_aprobado() OWNER TO postgres;

--
-- Name: validar_unica_materia_activa(); Type: FUNCTION; Schema: public; Owner: postgres
--

CREATE FUNCTION public.validar_unica_materia_activa() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
DECLARE
    materias_activas INT;
BEGIN
    SELECT COUNT(*) INTO materias_activas
    FROM inscripciones i
    WHERE i.id_estudiante = NEW.id_estudiante
    AND i.estado IN ('pendiente_matricula', 'activo')
    AND i.id_inscripcion != COALESCE(NEW.id_inscripcion, -1);
    
    IF materias_activas >= 1 THEN
        RAISE EXCEPTION 'El estudiante solo puede cursar UNA materia a la vez. Actualmente tiene % materia(s) activa(s)', 
                        materias_activas;
    END IF;
    
    RETURN NEW;
END;
$$;

--
-- Name: resumen_academico_estudiante; Type: VIEW; Schema: public; Owner: postgres
--

CREATE VIEW public.resumen_academico_estudiante AS
 SELECT e.id_estudiante,
    u.nombre,
    u.apellido,
    e.legajo,
    c.nombre AS carrera,
    count(
        CASE
            WHEN ((i.estado)::text = 'aprobado'::text) THEN 1
            ELSE NULL::integer
        END) AS cant_aprobadas,
    count(
        CASE
            WHEN ((i.estado)::text = 'reprobado'::text) THEN 1
            ELSE NULL::integer
        END) AS cant_reprobadas,
    count(
        CASE
            WHEN ((i.estado)::text = ANY ((ARRAY['activo'::character varying, 'pendiente_matricula'::character varying])::text[])) THEN 1
            ELSE NULL::integer
        END) AS cant_en_curso,
    count(
        CASE
            WHEN ((i.estado)::text = 'retirado'::text) THEN 1
            ELSE NULL::integer
        END) AS cant_retiradas,
    count(i.id_inscripcion) AS cant_total_inscriptas,
    round(avg(
        CASE
            WHEN ((i.estado)::text = 'aprobado'::text) THEN i.calificacion_final
            ELSE NULL::numeric
        END), 2) AS promedio_aprobadas,
    round(avg(
        CASE
            WHEN (i.calificacion_final IS NOT NULL) THEN i.calificacion_final
            ELSE NULL::numeric
        END), 2) AS promedio_general,
    round(((100.0 * (count(
        CASE
            WHEN ((i.estado)::text = 'aprobado'::text) THEN 1
            ELSE NULL::integer
        END))::numeric) / (NULLIF(count(
        CASE
            WHEN ((i.estado)::text = ANY ((ARRAY['aprobado'::character varying, 'reprobado'::character varying])::text[])) THEN 1
            ELSE NULL::integer
        END), 0))::numeric), 2) AS tasa_aprobacion
   FROM ((((public.estudiantes e
     JOIN public.usuarios u ON ((e.id_usuario = u.id_usuario)))
     LEFT JOIN public.afiliaciones_estudiante ae ON (((ae.id_estudiante = e.id_estudiante) AND ((ae.tipo_programa)::text = 'carrera'::text) AND ((ae.estado)::text = 'activo'::text))))
     LEFT JOIN public.carreras c ON ((c.id_carrera = ae.id_carrera)))
     LEFT JOIN public.inscripciones i ON ((i.id_estudiante = e.id_estudiante)))
  GROUP BY e.id_estudiante, u.nombre, u.apellido, e.legajo, c.nombre;


ALTER VIEW public.resumen_academico_estudiante OWNER TO postgres;

--
-- Name: v_progreso_carrera; Type: VIEW; Schema: public; Owner: postgres
--

CREATE VIEW public.v_progreso_carrera AS
 SELECT e.id_estudiante,
    (((u.nombre)::text || ' '::text) || (u.apellido)::text) AS estudiante,
    c.nombre AS carrera,
    count(DISTINCT
        CASE
            WHEN (i.aprobado = true) THEN oa.id_materia
            ELSE NULL::integer
        END) AS materias_aprobadas,
    count(DISTINCT mc.id_materia) AS total_materias_carrera,
    round(((100.0 * (count(DISTINCT
        CASE
            WHEN (i.aprobado = true) THEN oa.id_materia
            ELSE NULL::integer
        END))::numeric) / (NULLIF(count(DISTINCT mc.id_materia), 0))::numeric), 2) AS porcentaje_avance
   FROM ((((((public.estudiantes e
     JOIN public.usuarios u ON ((u.id_usuario = e.id_usuario)))
     LEFT JOIN public.afiliaciones_estudiante ae ON (((ae.id_estudiante = e.id_estudiante) AND ((ae.tipo_programa)::text = 'carrera'::text) AND ((ae.estado)::text = 'activo'::text))))
     LEFT JOIN public.carreras c ON ((c.id_carrera = ae.id_carrera)))
     LEFT JOIN public.malla_curricular mc ON ((mc.id_carrera = c.id_carrera)))
     LEFT JOIN public.inscripciones i ON ((i.id_estudiante = e.id_estudiante)))

--
-- Name: vista_historial_estudiante; Type: VIEW; Schema: public; Owner: postgres
--

CREATE VIEW public.vista_historial_estudiante AS
 SELECT se.id_seguimiento,
    e.id_estudiante,
    pd.nombre AS periodo_nombre,
    pd.fecha_inicio AS periodo_inicio,
    pd.fecha_fin AS periodo_fin,
    m.codigo AS materia_codigo,
    m.nombre AS materia_nombre,
    (((u_prof.nombre)::text || ' '::text) || (u_prof.apellido)::text) AS profesor_nombre,
    se.estado_actual,
        CASE se.estado_actual
            WHEN 'aprobado'::text THEN 'APROBADO'::text
            WHEN 'reprobado'::text THEN 'REPROBADO'::text
            ELSE 'EN CURSO'::text
        END AS estado_display,
    se.calificacion_final,
    1 AS numero_intento
   FROM (((((((public.seguimiento_estudiante se
     JOIN public.inscripciones i ON ((se.id_inscripcion = i.id_inscripcion)))
     JOIN public.estudiantes e ON ((i.id_estudiante = e.id_estudiante)))
     JOIN public.grupos g ON ((i.id_oferta = g.id_oferta)))
     JOIN public.materias m ON ((g.id_materia = m.id_materia)))
     JOIN public.periodos_dictado pd ON ((g.id_periodo = pd.id_periodo)))
     JOIN public.profesores p ON ((g.id_profesor = p.id_profesor)))
     JOIN public.usuarios u_prof ON ((p.id_usuario = u_prof.id_usuario)));
